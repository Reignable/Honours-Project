%Glossary
\newglossaryentry{fork}{
	name={Fork},
	text={fork},
	description={The front suspension unit on a mountain bike}
}

\newglossaryentry{shock}{
	name={Shock},
	text={shock},
	description={The rear suspension unit. Only found on full suspension mountain bikes}
}

\newglossaryentry{hardtail}{
	name={Hard Tail},
	text={hard tail},
	description={A mountain bike with only front suspension}
}

\newglossaryentry{fullsus}{
	name={Full Suspension},
	text={full suspension},
	description={A mountain bike with both front and rear suspension}
}

\newglossaryentry{rebounddamping}{
	name={Rebound Damping},
	text={rebound damping},
	description={Controls the speed at which a suspension unit extends once it has been compressed. Less damping means the unit extends faster}
}
